function getTodayIndex(){
 const today = new Date();
 const diff = Math.floor((today - startDate)/(1000*60*60*24));
 return diff >= 0 && diff < tidyTasks.length ? diff : 0;
}

function prevDay(){ if(currentIndex>0){ currentIndex--; renderDaily(currentIndex);} }

function generateCalendar(year,month){
 const cal=document.getElementById("calendar"); cal.innerHTML="";
 const wd=["日","一","二","三","四","五","六"];
 wd.forEach(w=>{let e=document.createElement("div");e.textContent=w; e.style.fontWeight="bold"; e.style.textAlign="center"; cal.appendChild(e);});
 let firstDay=new Date(`${year}-${String(month+1).padStart(2,'0')}-01T00:00:00`).getDay();
 let days=new Date(year,month+1,0).getDate();
 for(let i=0;i<firstDay;i++){cal.appendChild(document.createElement("div"));}
 for(let date=1;date<=days;date++){
  let box=document.createElement("div");
  let d=new Date(`${year}-${String(month+1).padStart(2,'0')}-${String(date).padStart(2,'0')}T00:00:00`);
  let idx=Math.floor((d-startDate)/(1000*60*60*24));
  if(idx>=0 && idx<tidyTasks.length){
    let task=tidyTasks[idx];
    let cyc=cycleData[idx % cycleData.length];
    box.innerHTML=`<strong>${date}</strong><br>${cyc.diet}<br>${task[1]}`;
    box.addEventListener("click",()=>{ currentIndex=idx; renderDaily(idx); showPage(2); });
  } else {
    box.textContent = date; box.style.opacity="0.4";
  }
  cal.appendChild(box);
 }
}

function showPage(n){

   document.getElementById("btn"+i).classList.remove("active");
 }
 document.getElementById("page"+n).classList.remove("hidden");
 document.getElementById("btn"+n).classList.add("active");
}
